Stylus Railtie

Railtie responsible for injecting stylus inside the Rails application and the Sprockets Asset Pipeline.

 
require 'stylus/tilt'

module Stylus
  class Railtie < ::Rails::Railtie

    initializer :setup_stylus do |app|
      config.app_generators.stylesheet_engine :styl
      next unless app.config.assets.enabled
